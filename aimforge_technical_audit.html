Ключові метрики телеметрії (createTargetTelemetry / finalizeTargetTelemetry):

2.2 Адаптивна система з динамічною складністю

Поточна система має статичні фази з фіксованою relativeSensitivity. Для професійного рівня потрібна реальна адаптація — складність має змінюватися в залежності від продуктивності гравця в межах однієї фази.

Архітектура Adaptive Difficulty Controller
Концепція: якщо гравець набирає >85% точності — збільшувати швидкість мішеней на 12%. Якщо <50% — зменшувати на 15%. Це дозволяє тримати гравця в "зоні потоку" (flow channel) протягом усього тесту. 
// src/game/AdaptiveDifficultyController.js
export class AdaptiveDifficultyController {
  constructor(baseScenario, targetAccuracy = 0.72, windowSize = 8) {
    this.baseScenario = baseScenario;
    this.targetAccuracy = targetAccuracy;
    this.windowSize = windowSize;
    this.accuracyWindow = [];
    this.reactionWindow = [];
    this.difficultyMultiplier = 1.0;
    this.minMultiplier = 0.5;
    this.maxMultiplier = 2.0;
    this.adaptationHistory = [];
  }

  feedEvent(isHit, reactionMs, targetAngularError) {
    this.accuracyWindow.push(isHit ? 1 : 0);
    if (reactionMs > 0) this.reactionWindow.push(reactionMs);
    if (this.accuracyWindow.length > this.windowSize) {
      this.accuracyWindow.shift();
      if (this.reactionWindow.length > this.windowSize) this.reactionWindow.shift();
    }
    return this.evaluateAndAdapt();
  }

  evaluateAndAdapt() {
    if (this.accuracyWindow.length < this.windowSize * 0.5) return null;

    const currentAccuracy = this.accuracyWindow.reduce((a, b) => a + b, 0) / this.accuracyWindow.length;
    const currentReaction = this.reactionWindow.length 
      ? this.reactionWindow.reduce((a, b) => a + b, 0) / this.reactionWindow.length 
      : 400;

    // PID-like controller for difficulty
    const accuracyError = this.targetAccuracy - currentAccuracy;
    const reactionNorm = Math.max(0, Math.min(1, (currentReaction - 150) / 500));

    // Higher accuracy → increase difficulty
    // Lower reaction time → can handle harder targets
    const adjustment = accuracyError * 0.3 + (reactionNorm - 0.5) * 0.15;
    this.difficultyMultiplier = clamp(
      this.difficultyMultiplier - adjustment,
      this.minMultiplier,
      this.maxMultiplier
    );

    const adaptedParams = this.computeAdaptedParams();
    this.adaptationHistory.push({
      timestamp: performance.now(),
      multiplier: this.difficultyMultiplier,
      accuracy: currentAccuracy,
      reaction: currentReaction,
    });

    return adaptedParams;
  }

  computeAdaptedParams() {
    const m = this.difficultyMultiplier;
    return {
      spawnInterval: this.baseScenario.spawnInterval / Math.sqrt(m),     // harder = faster spawn
      expiry: this.baseScenario.expiry / m,                               // harder = less time
      radius: this.baseScenario.radius / Math.sqrt(Math.min(m, 1.4)),     // harder = smaller targets
      movementSpeed: m,                                                    // harder = faster movement
    };
  }
}

2.3 Математична модель EDPI-оптимізації

Поточна рекомендація базується на weighted average по compositeScore. Для професійного рівня потрібна модель, яка враховує Pareto-оптимальність між швидкістю реакції та точністю.

Модель "EDPI Sweet Spot" (квадратична регресія):

// src/game/SensitivityOptimizer.js
export class SensitivityOptimizer {
  // Fit quadratic model: Score = a*sens^2 + b*sens + c
  // Find vertex (optimal) and confidence interval

  static analyze(results) {
    // results: Array<{ valorantSensitivity, compositeScore, accuracy, avgReaction }>
    if (results.length < 4) return null;

    // 1. Quadratic regression for composite score
    const n = results.length;
    const sumX = results.reduce((s, r) => s + r.valorantSensitivity, 0);
    const sumX2 = results.reduce((s, r) => s + r.valorantSensitivity ** 2, 0);
    const sumX3 = results.reduce((s, r) => s + r.valorantSensitivity ** 3, 0);
    const sumX4 = results.reduce((s, r) => s + r.valorantSensitivity ** 4, 0);
    const sumY = results.reduce((s, r) => s + r.compositeScore, 0);
    const sumXY = results.reduce((s, r) => s + r.valorantSensitivity * r.compositeScore, 0);
    const sumX2Y = results.reduce((s, r) => s + r.valorantSensitivity ** 2 * r.compositeScore, 0);

    // Solve normal equations for [a, b, c] in y = ax² + bx + c
    const matrix = [
      [sumX4, sumX3, sumX2],
      [sumX3, sumX2, sumX],
      [sumX2, sumX,  n],
    ];
    const rhs = [sumX2Y, sumXY, sumY];
    const [a, b, c] = solve3x3(matrix, rhs);

    // 2. Vertex = optimal sensitivity
    const optimalSens = -b / (2 * a);
    const optimalScore = a * optimalSens ** 2 + b * optimalSens + c;

    // 3. Confidence interval: where score >= 95% of optimal
    const threshold = optimalScore * 0.95;
    const discriminant = b ** 2 - 4 * a * (c - threshold);
    let rangeMin, rangeMax;
    if (discriminant >= 0) {
      rangeMin = (-b - Math.sqrt(discriminant)) / (2 * a);
      rangeMax = (-b + Math.sqrt(discriminant)) / (2 * a);
    }

    // 4. Reaction-Accuracy Pareto Frontier
    const pareto = this.computeParetoFrontier(results);

    // 5. Stability metric: lower corrections = more stable aim
    const stabilityScore = results.reduce((s, r) => {
      return s + (r.avgCorrections < 1.5 ? 1 : r.avgCorrections < 3 ? 0.5 : 0);
    }, 0) / results.length;

    return {
      recommendedValorantSensitivity: roundTo(optimalSens),
      recommendedSiteSensitivity: roundTo(valorantSensitivityToSite(dpi, optimalSens), 2),
      rangeMin: roundTo(rangeMin),
      rangeMax: roundTo(rangeMax),
      confidence: this.computeConfidence(a, results, optimalSens),
      stabilityScore,
      paretoFrontier: pareto,
      model: { a, b, c },
    };
  }

  static computeParetoFrontier(results) {
    // Find points where improving one metric worsens another
    const sorted = [...results].sort((a, b) => a.avgReaction - b.avgReaction);
    const frontier = [];
    let maxAccuracy = -1;
    for (const point of sorted) {
      if (point.accuracy > maxAccuracy) {
        frontier.push(point);
        maxAccuracy = point.accuracy;
      }
    }
    return frontier;
  }

  static computeConfidence(a, results, optimal) {
    // Negative a = concave down (good, has maximum)
    // Magnitude of a = how "sharp" the peak is
    if (a >= 0) return 0.35; // No clear optimum

    const curvature = Math.abs(a);
    const spread = results.reduce((s, r) => {
      return s + (r.valorantSensitivity - optimal) ** 2;
    }, 0) / results.length;

    return clamp(0.5 + curvature * 10 - spread * 2, 0.35, 0.96);
  }
}

3. Оптимізація Рушія

3.1 InstancedMesh для частинок і мішеней

Поточний ParticlePool створює N окремих Mesh. При 64 частинках це прийнятно, але для ефектів типу "hit spark explosion" (200+ частинок) потрібен InstancedMesh.

Переваги InstancedMesh
  • 1 draw call для 1000+ екземплярів замість 1000 draw calls
  • Матриця трансформації обчислюється на CPU, передається як атрибут інстансу
  • Гарантована сумісність з frustum culling (per-instance)
// src/game/InstancedParticleSystem.js
import * as THREE from 'three';

export class InstancedParticleSystem {
  constructor(scene, maxCount = 512) {
    this.scene = scene;
    this.maxCount = maxCount;
    this.count = 0;

    // Single geometry + material for ALL particles
    const geometry = new THREE.SphereGeometry(0.03, 6, 6);
    const material = new THREE.MeshBasicMaterial({
      color: 0xff6644,
      transparent: true,
      opacity: 1,
      depthWrite: false,
    });

    this.mesh = new THREE.InstancedMesh(geometry, material, maxCount);
    this.mesh.instanceMatrix.setUsage(THREE.DynamicDrawUsage);
    this.mesh.frustumCulled = false; // Particles are short-lived, skip culling
    scene.add(this.mesh);

    // Particle state arrays (object pool pattern)
    this.active = new Uint8Array(maxCount);
    this.velocities = new Float32Array(maxCount * 3);
    this.lifetimes = new Float32Array(maxCount);
    this.maxLifetimes = new Float32Array(maxCount);
    this.colors = new Float32Array(maxCount * 3);

    this.dummy = new THREE.Object3D();
    this._freeIndices = [];
    for (let i = maxCount - 1; i >= 0; i--) this._freeIndices.push(i);
  }

  burst(position, color = '#ff6644', count = 12) {
    const c = new THREE.Color(color);
    for (let i = 0; i < count && this._freeIndices.length > 0; i++) {
      const idx = this._freeIndices.pop();
      this.active[idx] = 1;

      // Velocity with gravity-aware spread
      this.velocities[idx * 3] = (Math.random() - 0.5) * 4;
      this.velocities[idx * 3 + 1] = Math.random() * 3 + 0.5;
      this.velocities[idx * 3 + 2] = (Math.random() - 0.5) * 4;

      this.lifetimes[idx] = 0.4 + Math.random() * 0.3;
      this.maxLifetimes[idx] = this.lifetimes[idx];

      this.colors[idx * 3] = c.r;
      this.colors[idx * 3 + 1] = c.g;
      this.colors[idx * 3 + 2] = c.b;

      this.dummy.position.copy(position);
      this.dummy.updateMatrix();
      this.mesh.setMatrixAt(idx, this.dummy.matrix);
    }
    this.mesh.instanceMatrix.needsUpdate = true;
  }

  update(delta) {
    let needsUpdate = false;
    const gravity = -9.8;

    for (let i = 0; i < this.maxCount; i++) {
      if (!this.active[i]) continue;

      this.lifetimes[i] -= delta;
      if (this.lifetimes[i] <= 0) {
        this.active[i] = 0;
        this._freeIndices.push(i);
        // Move off-screen
        this.dummy.position.set(0, -9999, 0);
        this.dummy.updateMatrix();
        this.mesh.setMatrixAt(i, this.dummy.matrix);
        needsUpdate = true;
        continue;
      }

      // Update velocity with gravity
      this.velocities[i * 3 + 1] += gravity * delta;

      // Get current position from matrix
      this.mesh.getMatrixAt(i, this.dummy.matrix);
      this.dummy.position.setFromMatrixPosition(this.dummy.matrix);

      // Apply velocity
      this.dummy.position.x += this.velocities[i * 3] * delta;
      this.dummy.position.y += this.velocities[i * 3 + 1] * delta;
      this.dummy.position.z += this.velocities[i * 3 + 2] * delta;

      // Scale down as life fades
      const lifeRatio = this.lifetimes[i] / this.maxLifetimes[i];
      this.dummy.scale.setScalar(lifeRatio);

      this.dummy.updateMatrix();
      this.mesh.setMatrixAt(i, this.dummy.matrix);
      needsUpdate = true;
    }

    if (needsUpdate) {
      this.mesh.instanceMatrix.needsUpdate = true;
    }
  }

  dispose() {
    this.scene.remove(this.mesh);
    this.mesh.geometry.dispose();
    this.mesh.material.dispose();
  }
}

      emissive: 0xffffff,
      emissiveIntensity: 0.3,
      roughness: 0.45,
      metalness: 0.08,
    });

    this.mesh = new THREE.InstancedMesh(geometry, material, maxTargets);
    this.mesh.instanceMatrix.setUsage(THREE.DynamicDrawUsage);
    this.mesh.instanceColor = new THREE.InstancedBufferAttribute(
      new Float32Array(maxTargets * 3), 3
    );
    scene.add(this.mesh);

    this.targetData = new Array(maxTargets).fill(null);
    this.availableSlots = [];
    for (let i = maxTargets - 1; i >= 0; i--) this.availableSlots.push(i);

    this.dummy = new THREE.Object3D();
  }

  spawnTarget(position, radius = 0.3, color = '#ff6644', metadata = {}) {
    if (this.availableSlots.length === 0) return null;
    const slot = this.availableSlots.pop();

    this.dummy.position.copy(position);
    this.dummy.scale.setScalar(radius);
    this.dummy.updateMatrix();

    this.mesh.setMatrixAt(slot, this.dummy.matrix);

    const c = new THREE.Color(color);
    this.mesh.setColorAt(slot, c);

    this.targetData[slot] = {
      active: true,
      position: position.clone(),
      radius,
      spawnTime: performance.now(),
      ...metadata,
    };

    this.mesh.instanceMatrix.needsUpdate = true;
    if (this.mesh.instanceColor) this.mesh.instanceColor.needsUpdate = true;

    return slot; // Return instance ID for raycasting
  }

  updateTarget(slot, newPosition) {
    if (!this.targetData[slot]?.active) return;

    this.targetData[slot].position.copy(newPosition);
    this.mesh.getMatrixAt(slot, this.dummy.matrix);
    this.dummy.position.copy(newPosition);
    this.dummy.updateMatrix();
    this.mesh.setMatrixAt(slot, this.dummy.matrix);
    this.mesh.instanceMatrix.needsUpdate = true;
  }

  removeTarget(slot) {
    if (!this.targetData[slot]?.active) return;
    this.targetData[slot].active = false;
    this.availableSlots.push(slot);

    // Move off-screen
    this.dummy.position.set(0, -9999, 0);
    this.dummy.updateMatrix();
    this.mesh.setMatrixAt(slot, this.dummy.matrix);
    this.mesh.instanceMatrix.needsUpdate = true;
  }

  // Optimized raycast: test bounding spheres first, then precise
  raycast(ray) {
    const hits = [];
    for (let i = 0; i < this.maxTargets; i++) {
      if (!this.targetData[i]?.active) continue;
      const target = this.targetData[i];
      const distSq = ray.distanceSqToPoint(target.position);
      if (distSq <= target.radius * target.radius) {
        const intersection = new THREE.Vector3();
        ray.intersectSphere(
          new THREE.Sphere(target.position, target.radius),
          intersection
        );
        hits.push({ instanceId: i, point: intersection, distance: ray.origin.distanceTo(intersection) });
      }
    }
    hits.sort((a, b) => a.distance - b.distance);
    return hits;
  }

  dispose() {
    this.scene.remove(this.mesh);
    this.mesh.geometry.dispose();
    this.mesh.material.dispose();
  }
}

      this.updateBotAnimation(bot, delta);

      // Auto-release if expired
      if (bot.userData.expiryAt && now > bot.userData.expiryAt) {

      }
    }

  updateBotBehavior(bot, delta, camera, scenario, now) {
    const behavior = bot.userData.behavior;
    const age = (now - bot.userData.spawnTime) / 1000;
    const toCamera = camera.position.clone().sub(bot.position).normalize();

    switch (behavior) {
      case 'idle': {
        // Face camera with slight breathing
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        bot.position.y += Math.sin(age * 2) * 0.001;
        break;
      }
      case 'strafe': {
        // Horizontal strafe with sine wave
        const strafeSpeed = 1.5;
        const strafeRange = 4;
        const strafeOffset = Math.sin(age * strafeSpeed) * strafeRange;
        bot.position.x = bot.userData.spawnPosition.x + strafeOffset;
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        break;
      }
      case 'approach': {
        // Walk toward camera
        const approachSpeed = 1.2;
        bot.position.add(toCamera.multiplyScalar(approachSpeed * delta));
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        // Animate legs
        this.animateWalk(bot, age, 8);
        break;
      }
      case 'rush': {
        // Fast approach with slight jitter
        const rushSpeed = 3.5;
        const jitter = Math.sin(age * 15) * 0.3;
        toCamera.x += jitter * delta;
        bot.position.add(toCamera.normalize().multiplyScalar(rushSpeed * delta));
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        this.animateWalk(bot, age, 12);
        break;
      }
      case 'wide-strafe': {
        // Wide strafe at distance
        const speed = 2.0;
        const strafeX = Math.sin(age * speed) * 6;
        const strafeZ = Math.cos(age * speed * 0.7) * 3;
        bot.position.set(
          bot.userData.spawnPosition.x + strafeX,
          bot.userData.spawnPosition.y,
          bot.userData.spawnPosition.z + strafeZ
        );
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        break;
      }
      case 'peek': {
        // Peek from cover and hide
        const cycle = 3; // seconds per peek cycle
        const phase = (age % cycle) / cycle;
        if (phase < 0.3) {
          // Emerging
          bot.position.z = THREE.MathUtils.lerp(
            bot.userData.spawnPosition.z - 2,
            bot.userData.spawnPosition.z,
            phase / 0.3
          );
        } else if (phase < 0.6) {
          // Exposed
          bot.position.z = bot.userData.spawnPosition.z;
        } else {
          // Hiding
          bot.position.z = THREE.MathUtils.lerp(
            bot.userData.spawnPosition.z,
            bot.userData.spawnPosition.z - 2,
            (phase - 0.6) / 0.4
          );
        }
        bot.rotation.y = Math.atan2(toCamera.x, toCamera.z);
        break;
      }
    }
  }

  animateWalk(bot, age, frequency) {
    const parts = bot.userData.parts;
    if (!parts) return;
    const swing = Math.sin(age * frequency) * 0.15;
    parts.leftUpperLeg.rotation.x = swing;
    parts.rightUpperLeg.rotation.x = -swing;
    parts.leftLowerLeg.rotation.x = Math.max(0, swing * 0.5);
    parts.rightLowerLeg.rotation.x = Math.max(0, -swing * 0.5);
  }

  animateBotAnimation(bot, delta) {
    // Subtle idle breathing animation
    const parts = bot.userData.parts;
    if (!parts) return;
    const breath = Math.sin(performance.now() * 0.003) * 0.02;
    parts.torso.position.y = parts.torso.userData.defaultPosition.y + breath;
    parts.neck.position.y = parts.neck.userData.defaultPosition.y + breath * 0.5;
  }

  getActiveBots() {
    return Array.from(this.activeBots);
  }

  clear() {
    for (const bot of this.activeBots) {
      this.release(bot);
    }
  }

  dispose() {
    this.clear();
    for (const bot of this.pool) {
      this.scene.remove(bot);
      // Dispose all geometries and materials
      bot.traverse((child) => {
        if (child.isMesh) {
          child.geometry.dispose();
          child.material.dispose();
        }
      });
    }
  }

Поточний Raycaster (GameEngine.js) перебирає ВСІ об'єкти сцени. Для складних моделей ботів це O(n) на кожен постріл. Рішення — spatial partitioning та hierarchical raycasting.

// src/game/SpatialIndex.js — Grid-based spatial hashing
export class SpatialIndex {
  constructor(cellSize = 4) {
    this.cellSize = cellSize;
    this.grid = new Map(); // key: "x,y,z", value: Set of objects
  }

  _getCellKey(x, y, z) {
    const cx = Math.floor(x / this.cellSize);
    const cy = Math.floor(y / this.cellSize);
    const cz = Math.floor(z / this.cellSize);
    return `${cx},${cy},${cz}`;
  }

  insert(obj, position) {
    const key = this._getCellKey(position.x, position.y, position.z);
    if (!this.grid.has(key)) this.grid.set(key, new Set());
    this.grid.get(key).add(obj);
    obj._spatialKey = key;
  }

  update(obj, newPosition) {
    this.remove(obj);
    this.insert(obj, newPosition);
  }

  remove(obj) {
    if (obj._spatialKey && this.grid.has(obj._spatialKey)) {
      this.grid.get(obj._spatialKey).delete(obj);
    }
  }

  // Get candidates along ray using 3D DDA traversal
  queryRay(ray, maxDistance = 50) {
    const candidates = new Set();
    const dir = ray.direction.clone().normalize();
    const invDir = new THREE.Vector3(1/dir.x, 1/dir.y, 1/dir.z);

    // Step through grid cells along ray
    let t = 0;
    const step = new THREE.Vector3(
      Math.sign(dir.x), Math.sign(dir.y), Math.sign(dir.z)
    );
    const tDelta = new THREE.Vector3(
      Math.abs(this.cellSize * invDir.x),
      Math.abs(this.cellSize * invDir.y),
      Math.abs(this.cellSize * invDir.z)
    );

    const cell = new THREE.Vector3(
      Math.floor(ray.origin.x / this.cellSize),
      Math.floor(ray.origin.y / this.cellSize),
      Math.floor(ray.origin.z / this.cellSize)
    );

    const tMax = new THREE.Vector3(
      ((cell.x + (step.x > 0 ? 1 : 0)) * this.cellSize - ray.origin.x) * invDir.x,
      ((cell.y + (step.y > 0 ? 1 : 0)) * this.cellSize - ray.origin.y) * invDir.y,
      ((cell.z + (step.z > 0 ? 1 : 0)) * this.cellSize - ray.origin.z) * invDir.z
    );

    let steps = 0;
    const maxSteps = Math.ceil(maxDistance / this.cellSize) * 3;

    while (t < maxDistance && steps < maxSteps) {
      const key = `${cell.x},${cell.y},${cell.z}`;
      if (this.grid.has(key)) {
        for (const obj of this.grid.get(key)) {
          candidates.add(obj);
        }
      }

      // Step to next cell
      if (tMax.x < tMax.y) {
        if (tMax.x < tMax.z) {
          t = tMax.x;
          tMax.x += tDelta.x;
          cell.x += step.x;
        } else {
          t = tMax.z;
          tMax.z += tDelta.z;
          cell.z += step.z;
        }
      } else {
        if (tMax.y < tMax.z) {
          t = tMax.y;
          tMax.y += tDelta.y;
          cell.y += step.y;
        } else {
          t = tMax.z;
          tMax.z += tDelta.z;
          cell.z += step.z;
        }
      }
      steps++;
    }

    return candidates;
  }

  clear() {
    this.grid.clear();
  }
}

// Optimized raycasting in GameEngine
class OptimizedGameEngine {
  constructor() {
    this.spatialIndex = new SpatialIndex(4); // 4-meter cells
  }

  performRaycast() {
    const ray = new THREE.Raycaster();
    ray.setFromCamera(new THREE.Vector2(0, 0), this.camera);

    // Query spatial index for candidates
    const candidates = this.spatialIndex.queryRay(ray.ray, 60);

    // Test only candidates (not entire scene)
    const targets = [];
    for (const obj of candidates) {
      if (obj.isInstancedMesh) {
        // Special handling for InstancedMesh
        const hits = this.raycastInstancedMesh(ray, obj);
        targets.push(...hits);
      } else if (obj.isGroup) {
        // For bot groups, test hitboxes only
        const hit = this.raycastBot(ray, obj);
        if (hit) targets.push(hit);
      }
    }

    targets.sort((a, b) => a.distance - b.distance);
    return targets[0] || null;
  }

  raycastBot(ray, botGroup) {
    // Only test hitbox meshes, not full geometry
    const hitboxes = botGroup.userData.hitboxes || [];
    let closestHit = null;
    let closestDist = Infinity;

    for (const mesh of hitboxes) {
      if (!mesh.visible) continue;
      const intersects = ray.intersectObject(mesh, false);
      if (intersects.length > 0 && intersects[0].distance < closestDist) {
        closestDist = intersects[0].distance;
        closestHit = {
          ...intersects[0],
          hitPart: mesh.userData.hitPart || 'body',
          bot: botGroup,
        };
      }
    }

    return closestHit;
  }
}

4. Flick Path Analysis

Система візуалізації "шляху прицілу" (aim path) дозволяє аналізувати якість фліків після кожного пострілу. Ключові метрики: overshoot, undershoot, correction count, path smoothness.

// src/game/FlickPathAnalyzer.js
import * as THREE from 'three';

export class FlickPathAnalyzer {
  constructor(scene, maxPathLength = 200) {
    this.scene = scene;
    this.maxPathLength = maxPathLength;

    // Recording state
    this.isRecording = false;
    this.currentPath = []; // Array of {timestamp, yaw, pitch, angularVelocity}
    this.pathHistory = []; // Array of completed flick paths

    // Visualization
    this.pathLine = null;
    this.idealLine = null;
    this.markerGroup = new THREE.Group();
    scene.add(this.markerGroup);

    this.maxHistory = 10;
    this.lookPointer = { yaw: 0, pitch: 0 };
    this.lastPointer = null;
  }

  // Call this every frame from GameEngine.update()
  recordFrame(pointer, targetPosition, camera) {
    const now = performance.now();

    if (!this.isRecording) return;

    // Compute angular velocity
    let angularVel = 0;
    if (this.lastPointer) {
      const yawDelta = Math.abs(this.normalizeAngle(pointer.yaw - this.lastPointer.yaw));
      const pitchDelta = Math.abs(pointer.pitch - this.lastPointer.pitch);
      angularVel = Math.sqrt(yawDelta * yawDelta + pitchDelta * pitchDelta);
    }

    // Compute angular error to target
    const angularError = this.computeAngularError(pointer, targetPosition, camera);

    this.currentPath.push({
      timestamp: now,
      yaw: pointer.yaw,
      pitch: pointer.pitch,
      angularVelocity: angularVel,
      angularError,
    });

    // Keep only last N frames
    if (this.currentPath.length > this.maxPathLength) {
      this.currentPath.shift();
    }

    this.lastPointer = { yaw: pointer.yaw, pitch: pointer.pitch };
  }

  startFlick(startPointer, targetPosition, camera) {
    this.isRecording = true;
    this.currentPath = [];
    this.lastPointer = null;
    this.flickStartTime = performance.now();
    this.flickTarget = targetPosition.clone();
    this.initialError = this.computeAngularError(startPointer, targetPosition, camera);
  }

  endFlick(hitSuccess) {
    this.isRecording = false;

    if (this.currentPath.length < 3) return;

    const analysis = this.analyzePath(this.currentPath, hitSuccess);
    this.pathHistory.push({
      path: [...this.currentPath],
      analysis,
      hitSuccess,
      timestamp: performance.now(),
    });

    if (this.pathHistory.length > this.maxHistory) {
      this.pathHistory.shift();
    }

    this.visualizeLastPath();
    return analysis;
  }

  analyzePath(path, hitSuccess) {
    if (path.length < 3) return null;

    const errors = path.map(p => p.angularError);
    const velocities = path.map(p => p.angularVelocity);

    // Find minimum error (closest to target)
    const minError = Math.min(...errors);
    const minErrorIndex = errors.indexOf(minError);

    // Overshoot detection: did we pass the target and come back?
    let overshootAmount = 0;
    let correctionCount = 0;
    let lastDirection = 0;

    for (let i = 1; i < path.length; i++) {
      const errorDelta = errors[i] - errors[i - 1];
      const direction = Math.sign(errorDelta);
      if (direction !== 0 && direction !== lastDirection && lastDirection !== 0) {
        correctionCount++;
        // If we're getting closer after getting farther, track overshoot
        if (errors[i - 1] > errors[0] * 0.5) {
          overshootAmount = Math.max(overshootAmount, errors[i - 1] - minError);
        }
      }
      if (direction !== 0) lastDirection = direction;
    }

    // Smoothness: standard deviation of angular acceleration
    const accelerations = [];
    for (let i = 1; i < velocities.length; i++) {
      accelerations.push(velocities[i] - velocities[i - 1]);
    }
    const avgAccel = accelerations.reduce((a, b) => a + b, 0) / accelerations.length;
    const accelVariance = accelerations.reduce((sum, a) => sum + (a - avgAccel) ** 2, 0) / accelerations.length;
    const smoothnessScore = 1 / (1 + Math.sqrt(accelVariance) * 10); // 0-1

    // Time to first zero-crossing (first time we get within small error)
    const threshold = 0.05; // radians
    let timeToAcquire = null;
    for (let i = 0; i < errors.length; i++) {
      if (errors[i] < threshold) {
        timeToAcquire = path[i].timestamp - path[0].timestamp;
        break;
      }
    }

    return {
      overshootAmount,
      overshootRatio: this.initialError > 0 ? overshootAmount / this.initialError : 0,
      correctionCount,
      smoothnessScore,
      timeToAcquire,
      minError,
      initialError: this.initialError,
      pathEfficiency: this.initialError > 0 
        ? this.initialError / this.computePathLength(path) 
        : 1,
      avgVelocity: velocities.reduce((a, b) => a + b, 0) / velocities.length,
      hitSuccess,
    };
  }

  computePathLength(path) {
    let length = 0;
    for (let i = 1; i < path.length; i++) {
      const dy = this.normalizeAngle(path[i].yaw - path[i - 1].yaw);
      const dp = path[i].pitch - path[i - 1].pitch;
      length += Math.sqrt(dy * dy + dp * dp);
    }
    return length;
  }

  // Visualize path as 3D line projected from camera
  visualizeLastPath() {
    // Clear previous visualization
    while (this.markerGroup.children.length > 0) {
      const child = this.markerGroup.children[0];
      if (child.geometry) child.geometry.dispose();
      if (child.material) child.material.dispose();
      this.markerGroup.remove(child);
    }

    if (this.pathHistory.length === 0) return;

    const lastPath = this.pathHistory[this.pathHistory.length - 1];
    const path = lastPath.path;

    // Convert angular coordinates to 3D points at fixed distance
    const distance = 5;
    const points = path.map(p => {
      const x = distance * Math.sin(p.yaw) * Math.cos(p.pitch);
      const y = distance * Math.sin(p.pitch);
      const z = -distance * Math.cos(p.yaw) * Math.cos(p.pitch);
      return new THREE.Vector3(x, y, z);
    });

    // Create path line with color gradient (red = start, green = end)
    const colors = points.map((_, i) => {
      const t = i / points.length;
      return new THREE.Color(1 - t, t, 0); // Red → Yellow → Green
    });

    const geometry = new THREE.BufferGeometry().setFromPoints(points);
    geometry.setAttribute('color', new THREE.Float32BufferAttribute(
      colors.flatMap(c => [c.r, c.g, c.b]), 3
    ));

    const material = new THREE.LineBasicMaterial({
      vertexColors: true,
      linewidth: 2,
      transparent: true,
      opacity: 0.8,
    });

    this.pathLine = new THREE.Line(geometry, material);
    this.markerGroup.add(this.pathLine);

    // Add overshoot markers
    const analysis = lastPath.analysis;
    if (analysis.overshootAmount > 0.02) {
      const overshootGeom = new THREE.SphereGeometry(0.08, 8, 8);
      const overshootMat = new THREE.MeshBasicMaterial({
        color: 0xff4444,
        transparent: true,
        opacity: 0.6,
      });
      const overshootMarker = new THREE.Mesh(overshootGeom, overshootMat);
      // Position at peak overshoot point
      const peakIdx = path.findIndex(p => p.angularError === analysis.minError + analysis.overshootAmount);
      if (peakIdx >= 0) {
        overshootMarker.position.copy(points[peakIdx]);
        this.markerGroup.add(overshootMarker);
      }
    }

    // Fade out after 3 seconds
    setTimeout(() => {
      this.fadeOutVisualization();
    }, 3000);
  }

  fadeOutVisualization() {
    if (!this.pathLine) return;
    const mat = this.pathLine.material;
    let opacity = 0.8;
    const fade = setInterval(() => {
      opacity -= 0.05;
      mat.opacity = Math.max(0, opacity);
      if (opacity <= 0) {
        clearInterval(fade);
        this.markerGroup.clear();
        this.pathLine = null;
      }
    }, 50);
  }

  computeAngularError(pointer, targetPos, camera) {
    const offset = targetPos.clone().sub(camera.position);
    const targetYaw = Math.atan2(offset.x, -offset.z);
    const targetPitch = Math.atan2(offset.y, Math.hypot(offset.x, offset.z));
    return Math.hypot(
      this.normalizeAngle(targetYaw - pointer.yaw),
      targetPitch - pointer.pitch
    );
  }

  normalizeAngle(angle) {
    return Math.atan2(Math.sin(angle), Math.cos(angle));
  }

  dispose() {
    this.markerGroup.clear();
    this.scene.remove(this.markerGroup);
  }

export class RecoilSystem {
  constructor(camera) {
    this.camera = camera;
    this.shotIndex = 0;          // Current position in pattern
    this.accumulatedRecoil = new THREE.Vector2(0, 0); // yaw, pitch offset
    this.currentDispersion = new THREE.Vector2(0, 0);
    this.isRecovering = false;
    this.lastShotTime = 0;
    this.currentPattern = null;
  }

  setWeapon(weaponType) {
    this.currentPattern = RECOIL_PATTERNS[weaponType] || RECOIL_PATTERNS.rifle;
    this.shotIndex = 0;
    this.accumulatedRecoil.set(0, 0);
  }

  fire(isADS = false) {
    if (!this.currentPattern) return new THREE.Vector2(0, 0);

    const pattern = this.currentPattern;
    const idx = Math.min(this.shotIndex, pattern.vertical.length - 1);

    // Base pattern value
    let verticalRecoil = pattern.vertical[idx] || 0;
    let horizontalRecoil = pattern.horizontal[idx] || 0;

    // Apply ADS reduction
    if (isADS) {
      verticalRecoil *= pattern.adsMultiplier;
      horizontalRecoil *= pattern.adsMultiplier;
    }

    // Add random spread (Gaussian-like distribution)
    const randomYaw = (Math.random() - 0.5) * 2 * pattern.randomSpread;
    const randomPitch = (Math.random() - 0.5) * 2 * pattern.randomSpread;

    verticalRecoil += randomPitch;
    horizontalRecoil += randomYaw;

    // Accumulate
    this.accumulatedRecoil.x += horizontalRecoil; // yaw
    this.accumulatedRecoil.y += verticalRecoil;   // pitch

    this.shotIndex++;
    this.lastShotTime = performance.now();
    this.isRecovering = false;

    return new THREE.Vector2(horizontalRecoil, verticalRecoil);
  }

  update(delta, pointer) {
    if (!this.currentPattern) return;

    const timeSinceShot = (performance.now() - this.lastShotTime) / 1000;
    const pattern = this.currentPattern;

    // Auto-recovery after 0.15s of not shooting
    if (timeSinceShot > 0.15) {
      this.isRecovering = true;
    }

    if (this.isRecovering) {
      // Exponential recovery toward zero
      const recoveryRate = pattern.recoverySpeed * delta;
      this.accumulatedRecoil.x *= Math.max(0, 1 - recoveryRate * 0.5);
      this.accumulatedRecoil.y *= Math.max(0, 1 - recoveryRate);

      // Reset shot index when fully recovered
      if (Math.abs(this.accumulatedRecoil.y) < 0.01) {
        this.shotIndex = 0;
        this.accumulatedRecoil.set(0, 0);
        this.isRecovering = false;
      }
    }
  }

  reset() {
    this.shotIndex = 0;
    this.accumulatedRecoil.set(0, 0);
    this.isRecovering = false;
  }

  // Visual: apply recoil to weapon model
  applyToWeapon(weaponGroup, progress) {
    if (!weaponGroup) return;
    // Kick animation on weapon
    const kickBack = Math.sin(progress * Math.PI) * 0.03;
    const kickUp = Math.sin(progress * Math.PI) * 0.015;
    weaponGroup.position.z += kickBack;
    weaponGroup.rotation.x -= kickUp;
  }
}

Для більш плавних патернів (без дискретних кроків) можна інтерполювати між опорними точками кривої Безьє: 
function bezierInterpolation(controlPoints, t) {
  // De Casteljau's algorithm
  let points = controlPoints.map(p => ({...p}));
  for (let level = 1; level < points.length; level++) {
    for (let i = 0; i < points.length - level; i++) {
      points[i].y = (1 - t) * points[i].y + t * points[i + 1].y;
      points[i].x = (1 - t) * points[i].x + t * points[i + 1].x;
    }
  }
  return points[0];
}

6. Ranked System

Ранкова система від Bronze до Radiant потребує: (1) стабільної метрики навички, (2) нормалізації між сценаріями, (3) розподілу по процентилях.

// src/game/RankingSystem.js

const RANKS = [
  { name: 'Bronze',    tier: 1,  percentileThreshold: 0 },
  { name: 'Silver',    tier: 2,  percentileThreshold: 15 },
  { name: 'Gold',      tier: 3,  percentileThreshold: 35 },
  { name: 'Platinum',  tier: 4,  percentileThreshold: 55 },
  { name: 'Diamond',   tier: 5,  percentileThreshold: 75 },
  { name: 'Ascendant', tier: 6,  percentileThreshold: 88 },
  { name: 'Immortal',  tier: 7,  percentileThreshold: 96 },
  { name: 'Radiant',   tier: 8,  percentileThreshold: 99.5 },
];

export class RankingSystem {
  constructor() {
    // Per-scenario rating baselines (populated from global data)
    this.baselines = new Map(); // scenarioId -> { p50, p75, p90, p99 }
  }

  // Skill Rating Formula (ELO-like with scenario weights)
  calculateSkillRating(scenarioResults, globalPercentiles) {
    /*
      scenarioResults: Array<{
        scenarioId, score, accuracy, avgReaction, consistencyCV
      }>

      Composite Skill Score = weighted sum of normalized performance
      across all scenario categories.
    */

    const categoryWeights = {
      flicking:   0.18,
      tracking:   0.18,
      precision:  0.16,
      speed:      0.12,
      gridshot:   0.12,
      valorant:   0.14,
      switching:  0.10,
    };

    let totalWeight = 0;
    let weightedSum = 0;

    for (const result of scenarioResults) {
      const baseline = this.baselines.get(result.scenarioId);
      if (!baseline) continue;

      // Normalize score against percentile baselines
      const normalizedScore = this.normalizeAgainstBaseline(
        result.score, baseline
      );

      // Consistency bonus (lower CV = higher bonus)
      const consistencyBonus = Math.max(0, 1 - result.consistencyCV) * 0.1;

      // Speed bonus (faster reaction = higher)
      const reactionNorm = Math.max(0, Math.min(1, 
        (400 - result.avgReaction) / 250
      ));

      const category = getScenarioById(result.scenarioId)?.category || 'warmup';
      const weight = categoryWeights[category] || 0.05;

      weightedSum += (normalizedScore + consistencyBonus + reactionNorm * 0.05) * weight;
      totalWeight += weight;
    }

    const rawRating = totalWeight > 0 ? (weightedSum / totalWeight) * 1000 : 500;

    return {
      rating: Math.round(rawRating),
      rank: this.getRankFromRating(rawRating),
      percentile: this.estimatePercentile(rawRating, globalPercentiles),
      categoryBreakdown: this.computeCategoryBreakdown(scenarioResults),
    };
  }

  normalizeAgainstBaseline(score, baseline) {
    // Convert score to percentile using log-normal distribution assumption
    // baseline: { p50, p75, p90, p99 }
    if (score <= baseline.p50) {
      return 0.5 * (score / baseline.p50);
    } else if (score <= baseline.p75) {
      return 0.5 + 0.25 * ((score - baseline.p50) / (baseline.p75 - baseline.p50));
    } else if (score <= baseline.p90) {
      return 0.75 + 0.15 * ((score - baseline.p75) / (baseline.p90 - baseline.p75));
    } else {
      return 0.90 + 0.10 * Math.min(1, (score - baseline.p90) / (baseline.p99 - baseline.p90));
    }
  }

  getRankFromRating(rating) {
    const percentile = this.ratingToPercentile(rating);
    for (let i = RANKS.length - 1; i >= 0; i--) {
      if (percentile >= RANKS[i].percentileThreshold) {
        return RANKS[i];
      }
    }
    return RANKS[0];
  }

  ratingToPercentile(rating) {
    // Logistic mapping: typical range 300-1200
    return 100 / (1 + Math.exp(-0.008 * (rating - 750)));
  }

  estimatePercentile(rating, globalDistribution) {
    if (!globalDistribution || globalDistribution.length === 0) return 50;

    // Binary search in sorted distribution
    const below = globalDistribution.filter(r => r < rating).length;
    return (below / globalDistribution.length) * 100;
  }

  computeCategoryBreakdown(results) {
    const categories = {};
    for (const result of results) {
      const cat = getScenarioById(result.scenarioId)?.category || 'other';
      if (!categories[cat]) categories[cat] = { scores: [], best: 0 };
      categories[cat].scores.push(result.score);
      categories[cat].best = Math.max(categories[cat].best, result.score);
    }

    return Object.entries(categories).map(([cat, data]) => ({
      category: cat,
      average: data.scores.reduce((a, b) => a + b, 0) / data.scores.length,
      best: data.best,
      count: data.scores.length,
    }));
  }

  // ELO-like update after each ranked scenario
  updateRating(currentRating, scenarioScore, expectedScore, kFactor = 32) {
    const actual = scenarioScore; // 0-1
    const expected = 1 / (1 + Math.pow(10, (expectedScore - currentRating) / 400));
    const newRating = currentRating + kFactor * (actual - expected);
    return Math.round(newRating);
  }
}

// Percentile calculation for global leaderboard
export function calculatePercentile(score, allScores) {
  if (!allScores.length) return 50;
  const sorted = [...allScores].sort((a, b) => a - b);
  const below = sorted.filter(s => s < score).length;
  return (below / sorted.length) * 100;
}

// src/audio/SpatialAudioManager.js
import * as THREE from 'three';

export class SpatialAudioManager {
  constructor(camera, masterVolume = 1.0) {
    this.camera = camera;
    this.masterVolume = masterVolume;
    this.listener = new THREE.AudioListener();
    camera.add(this.listener);

    this.sounds = new Map();
    this.loadedBuffers = new Map();
    this.loader = new THREE.AudioLoader();
  }

  async preloadSound(name, url) {
    return new Promise((resolve, reject) => {
      this.loader.load(url, (buffer) => {
        this.loadedBuffers.set(name, buffer);
        resolve(buffer);
      }, undefined, reject);
    });
  }

  createPositionalSound(name, position, options = {}) {
    const buffer = this.loadedBuffers.get(name);
    if (!buffer) {
      console.warn(`Sound "${name}" not preloaded`);
      return null;
    }

    const sound = new THREE.PositionalAudio(this.listener);
    sound.setBuffer(buffer);
    sound.setRefDistance(options.refDistance || 8);
    sound.setRolloffFactor(options.rolloffFactor || 1.5);
    sound.setDistanceModel(options.distanceModel || 'inverse');
    sound.setVolume((options.volume || 1.0) * this.masterVolume);

    if (options.loop) sound.setLoop(true);

    // Attach to a mesh at target position
    const mesh = new THREE.Object3D();
    mesh.position.copy(position);
    mesh.add(sound);

    // Add to scene temporarily
    if (this.camera.parent) {
      this.camera.parent.add(mesh);
    }

    this.sounds.set(sound.id, { sound, mesh, createdAt: performance.now() });
    return sound;
  }

  // Play spatial target spawn sound (allows directional hearing)
  playTargetSpawn(position, targetType = 'sphere') {
    const soundName = targetType === 'bot' ? 'bot_spawn' : 'target_spawn';
    const sound = this.createPositionalSound(soundName, position, {
      volume: 0.6,
      refDistance: 10,
      rolloffFactor: 1.2,
    });
    if (sound) {
      sound.play();
      // Auto-cleanup after playback
      setTimeout(() => this.disposeSound(sound.id), 3000);
    }
  }

  // Ambient directional cue for off-screen targets
  playProximityWarning(nearbyTargets) {
    const now = performance.now();
    if (this._lastWarning && now - this._lastWarning < 500) return;

    for (const target of nearbyTargets) {
      if (target.isOnScreen) continue;

      // Create directional "whoosh" from target direction
      const angle = Math.atan2(
        target.position.x - this.camera.position.x,
        target.position.z - this.camera.position.z
      );

      // Position sound source at target but with exaggerated distance
      // for clearer directional perception
      const soundPos = new THREE.Vector3(
        this.camera.position.x + Math.sin(angle) * 5,
        this.camera.position.y,
        this.camera.position.z + Math.cos(angle) * 5
      );

      const sound = this.createPositionalSound('proximity_whoosh', soundPos, {
        volume: 0.3,
        refDistance: 3,
        rolloffFactor: 0.5,
      });
      if (sound) sound.play();
    }

    this._lastWarning = now;
  }

  // HRTF-enhanced footsteps for bot movement
  playFootstep(botPosition, intensity = 1.0) {
    const sound = this.createPositionalSound('footstep', botPosition, {
      volume: 0.4 * intensity,
      refDistance: 6,
      rolloffFactor: 2,
    });
    if (sound) {
      // Slight pitch variation for realism
      sound.detune = (Math.random() - 0.5) * 200; // ±100 cents
      sound.play();
    }
  }

  disposeSound(id) {
    const entry = this.sounds.get(id);
    if (!entry) return;
    entry.sound.stop();
    if (entry.mesh.parent) entry.mesh.parent.remove(entry.mesh);
    this.sounds.delete(id);
  }

  setMasterVolume(volume) {
    this.masterVolume = volume;
    for (const { sound } of this.sounds.values()) {
      // Update all active sounds
      sound.setVolume(sound.getVolume() * volume);
    }
  }

  dispose() {
    for (const { sound, mesh } of this.sounds.values()) {
      sound.stop();
      if (mesh.parent) mesh.parent.remove(mesh);
    }
    this.sounds.clear();
    this.loadedBuffers.clear();
    this.camera.remove(this.listener);
  }
}

// src/game/ReplaySystem.js
import * as THREE from 'three';

// Ring buffer for continuous recording with fixed memory budget
export class ReplayRecorder {
  constructor(maxDuration = 600, tickRate = 60) {
    this.maxTicks = maxDuration * tickRate;
    this.tickRate = tickRate;
    this.tickInterval = 1000 / tickRate;

    // Fixed-size arrays for zero-allocation recording
    this.positions = new Float32Array(this.maxTicks * 3);    // x, y, z
    this.rotations = new Float32Array(this.maxTicks * 2);    // yaw, pitch
    this.events = new Uint8Array(this.maxTicks);              // 0=none, 1=shot, 2=hit, 3=miss
    this.targetStates = [];                                   // Sparse array of target snapshots

    this.writeIndex = 0;
    this.isRecording = false;
    this.startTime = 0;
    this.tickCount = 0;

    // Delta compression for positions
    this.lastPosition = new THREE.Vector3();
    this.quantizationScale = 1000; // mm precision
  }

  startRecording() {
    this.isRecording = true;
    this.writeIndex = 0;
    this.tickCount = 0;
    this.startTime = performance.now();
    this.targetStates = [];
  }

  recordTick(camera, pointer, eventType = 0, targets = []) {
    if (!this.isRecording || this.writeIndex >= this.maxTicks) return;

    const idx = this.writeIndex;

    // Quantize position with delta from last
    const px = Math.round((camera.position.x - this.lastPosition.x) * this.quantizationScale);
    const py = Math.round((camera.position.y - this.lastPosition.y) * this.quantizationScale);
    const pz = Math.round((camera.position.z - this.lastPosition.z) * this.quantizationScale);

    this.positions[idx * 3] = px;
    this.positions[idx * 3 + 1] = py;
    this.positions[idx * 3 + 2] = pz;

    this.lastPosition.copy(camera.position);

    // Quantize rotation (yaw, pitch in milliradians)
    this.rotations[idx * 2] = Math.round(pointer.yaw * 1000);
    this.rotations[idx * 2 + 1] = Math.round(pointer.pitch * 1000);

    // Event
    this.events[idx] = eventType;

    // Sparse target snapshot (only when targets change)
    if (targets.length > 0 && (eventType > 0 || this.tickCount % 10 === 0)) {
      this.targetStates.push({
        tick: idx,
        targets: targets.map(t => ({
          id: t.id,
          x: Math.round(t.position.x * this.quantizationScale),
          y: Math.round(t.position.y * this.quantizationScale),
          z: Math.round(t.position.z * this.quantizationScale),
          active: t.active,
          hp: t.hp || 0,
        })),
      });
    }

    this.writeIndex++;
    this.tickCount++;
  }

  stopRecording() {
    this.isRecording = false;
    return this.exportReplay();
  }

  exportReplay() {
    return {
      version: 1,
      tickRate: this.tickRate,
      totalTicks: this.writeIndex,
      duration: this.tickCount / this.tickRate,
      positions: this.positions.slice(0, this.writeIndex * 3),
      rotations: this.rotations.slice(0, this.writeIndex * 2),
      events: this.events.slice(0, this.writeIndex),
      targetStates: this.targetStates,
      basePosition: {
        x: this.lastPosition.x,
        y: this.lastPosition.y,
        z: this.lastPosition.z,
      },
    };
  }

  // Estimated memory usage
  getMemoryUsage() {
    const bytes = this.positions.byteLength + 
                  this.rotations.byteLength + 
                  this.events.byteLength +
                  JSON.stringify(this.targetStates).length * 2;
    return {
      bytes,
      kb: (bytes / 1024).toFixed(1),
      mb: (bytes / 1024 / 1024).toFixed(2),
    };
  }
}

// Ghost playback system
export class GhostPlayback {
  constructor(scene, replayData) {
    this.scene = scene;
    this.replay = replayData;
    this.currentTick = 0;
    this.isPlaying = false;

    // Ghost camera representation (transparent)
    this.ghostCamera = new THREE.Group();

    // Ghost weapon
    const weaponGeom = new THREE.BoxGeometry(0.3, 0.2, 0.6);
    const weaponMat = new THREE.MeshBasicMaterial({
      color: 0x00ffff,
      transparent: true,
      opacity: 0.3,
      wireframe: true,
    });
    this.ghostWeapon = new THREE.Mesh(weaponGeom, weaponMat);
    this.ghostCamera.add(this.ghostWeapon);

    // Ghost crosshair trail
    const trailGeom = new THREE.BufferGeometry();
    const trailMat = new THREE.LineBasicMaterial({
      color: 0x00ffff,
      transparent: true,
      opacity: 0.15,
    });
    this.trail = new THREE.Line(trailGeom, trailMat);
    this.trailPoints = [];

    // Ghost hit markers
    this.hitMarkers = new THREE.Group();

    scene.add(this.ghostCamera);
    scene.add(this.trail);
    scene.add(this.hitMarkers);
  }

  start() {
    this.isPlaying = true;
    this.currentTick = 0;
    this.startTime = performance.now();
    this.trailPoints = [];
  }

  update(delta) {
    if (!this.isPlaying) return;

    // Advance tick based on elapsed time
    const elapsed = (performance.now() - this.startTime) / 1000;
    const targetTick = Math.floor(elapsed * this.replay.tickRate);

    if (targetTick >= this.replay.totalTicks) {
      this.isPlaying = false;
      return;
    }

    // Replay all ticks from current to target (catch-up)
    for (; this.currentTick <= targetTick; this.currentTick++) {
      this.replayTick(this.currentTick);
    }
  }

  replayTick(tick) {
    // Decompress position (delta + running sum)
    const basePos = this.replay.basePosition;
    let x = 0, y = 0, z = 0;
    for (let i = 0; i <= tick; i++) {
      x += this.replay.positions[i * 3] / this.replay.quantizationScale || 0;
      y += this.replay.positions[i * 3 + 1] / this.replay.quantizationScale || 0;
      z += this.replay.positions[i * 3 + 2] / this.replay.quantizationScale || 0;
    }

    const px = basePos.x + x;
    const py = basePos.y + y;
    const pz = basePos.z + z;

    const yaw = this.replay.rotations[tick * 2] / 1000;
    const pitch = this.replay.rotations[tick * 2 + 1] / 1000;

    // Update ghost camera
    this.ghostCamera.position.set(px, py, pz);
    this.ghostCamera.rotation.order = 'YXZ';
    this.ghostCamera.rotation.y = yaw;
    this.ghostCamera.rotation.x = pitch;

    // Update trail
    const distance = 3;
    const lookPoint = new THREE.Vector3(
      px + Math.sin(yaw) * Math.cos(pitch) * distance,
      py + Math.sin(pitch) * distance,
      pz - Math.cos(yaw) * Math.cos(pitch) * distance
    );

    if (tick % 3 === 0) { // Add point every 3 ticks for performance
      this.trailPoints.push(lookPoint);
      if (this.trailPoints.length > 300) this.trailPoints.shift();

      this.trail.geometry.setFromPoints(this.trailPoints);
      this.trail.geometry.attributes.position.needsUpdate = true;
    }

    // Handle events
    const event = this.replay.events[tick];
    if (event === 1) { // Shot
      this.showGhostShot(px, py, pz, yaw, pitch);
    } else if (event === 2) { // Hit
      this.showGhostHit(lookPoint);
    }
  }

  showGhostShot(x, y, z, yaw, pitch) {
    // Brief flash on weapon
    this.ghostWeapon.material.opacity = 0.8;
    setTimeout(() => {
      this.ghostWeapon.material.opacity = 0.3;
    }, 50);
  }

  showGhostHit(position) {
    const geom = new THREE.RingGeometry(0.05, 0.08, 8);
    const mat = new THREE.MeshBasicMaterial({
      color: 0x00ffff,
      transparent: true,
      opacity: 0.5,
      side: THREE.DoubleSide,
    });
    const marker = new THREE.Mesh(geom, mat);
    marker.position.copy(position);
    marker.lookAt(this.ghostCamera.position);
    this.hitMarkers.add(marker);

    // Fade out
    let opacity = 0.5;
    const fade = setInterval(() => {
      opacity -= 0.02;
      mat.opacity = opacity;
      if (opacity <= 0) {
        clearInterval(fade);
        this.hitMarkers.remove(marker);
        geom.dispose();
        mat.dispose();
      }
    }, 50);
  }

  dispose() {
    this.scene.remove(this.ghostCamera);
    this.scene.remove(this.trail);
    this.scene.remove(this.hitMarkers);
    this.ghostWeapon.geometry.dispose();
    this.ghostWeapon.material.dispose();
    this.trail.geometry.dispose();
    this.trail.material.dispose();
  }
}

9. High Refresh Rate та Delta Time

Критична проблема: при 240Hz фізика оновлюється в 4 рази частіше ніж при 60Hz. Без компенсації мішені рухаються швидше на високих частотах.

Проблема: Frame-rate Dependent Physics
Поточний код (типовий підхід): 
position.x += velocity.x;  // BAD: velocity per frame, not per second!
При 240Hz мішень переміститься в 4 рази далі за ту ж секунду.

// Fixed timestep physics with accumulator pattern (Gaffer on Games)

export class FixedTimestepLoop {
  constructor(updateFn, renderFn, fixedDt = 1/120) {
    this.updateFn = updateFn;    // physics/logic update
    this.renderFn = renderFn;    // render update
    this.fixedDt = fixedDt;       // 120Hz physics
    this.accumulator = 0;
    this.lastTime = performance.now();
    this.running = false;

    // Stats
    this.physicsSteps = 0;
    this.renderFrames = 0;
  }

  start() {
    this.running = true;
    this.lastTime = performance.now();
    this.loop();
  }

  stop() {
    this.running = false;
  }

  loop = () => {
    if (!this.running) return;

    const now = performance.now();
    let frameTime = (now - this.lastTime) / 1000;
    this.lastTime = now;

    // Prevent spiral of death (tab inactive, etc.)
    if (frameTime > 0.25) frameTime = 0.25;

    this.accumulator += frameTime;
    this.physicsSteps = 0;

    // Fixed timestep updates
    while (this.accumulator >= this.fixedDt) {
      this.updateFn(this.fixedDt); // dt is ALWAYS fixed
      this.accumulator -= this.fixedDt;
      this.physicsSteps++;
    }

    // Interpolation factor for smooth rendering
    const alpha = this.accumulator / this.fixedDt;
    this.renderFn(frameTime, alpha);
    this.renderFrames++;

    requestAnimationFrame(this.loop);
  };
}

fixedUpdate(dt) {
  if (this.paused || this.ended || this.disposed) return;

  this.actualElapsed += dt;

  // ALL movement MUST use dt
  this.updateWeapon(dt);
  this.updateTargets(dt);      // dt-compensated movement
  this.updateParticles(dt);
  this.updateBots(dt);
  this.updateRecoil(dt);
  this.updateScreenShake(dt);

  this.spawnTick(dt);

}

function updateTargetPosition(target, dt) {
  switch (target.movement) {
    case 'horizontal':
      target.position.x += target.speed * dt;  // speed in units/second
      break;
    case 'circle':
      target.angle += target.angularSpeed * dt; // radians/second
      target.position.x = Math.cos(target.angle) * target.radius;
      target.position.z = Math.sin(target.angle) * target.radius;
      break;
    case 'figure8':
      target.angle += target.angularSpeed * dt;
      target.position.x = Math.sin(target.angle) * target.amplitude;
      target.position.y = Math.sin(target.angle * 2) * target.amplitude * 0.5;
      break;
  }
}

// Interpolation for smooth rendering between physics steps

  for (const target of this.activeTargets) {
    if (target.previousPosition && target.visualMesh) {
      target.visualMesh.position.lerpVectors(
        target.previousPosition,
        target.position,
        alpha
      );
    }
  }
